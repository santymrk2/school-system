        this.total = safe(total);
        this.porcentaje = 0d;
    }

}

        String no = (nombre == null ? "" : nombre);
        this.nombreCompleto = (ap + " " + no).trim();
        this.presentes = safe(presentes);
        this.ausentes = safe(ausentes);
        this.tarde = safe(tarde);
        this.retiroAnticipado = safe(retiroAnticipado);
        this.total = safe(total);
